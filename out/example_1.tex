% Use \newif\if{{<number>}} to define boolean flags for each cumulative level,
% and explicitly set them all to false using \{{<number>}}false
\newif\ifb % 2nd level, includes base contents
\newif\ifa % 3rd level, includes 2nd level contents and base contents...
\bfalse
\afalse

\title{Example}
\begin{document}
	\maketitle
	\begin{enumerate}
		\item This item is part of the base contents of the document.\ifb{ This sentence is only shown at levels 2 and above.}\fi
		\ifb
			\item This item is only shown at levels 2 and above.\ifa{ This sentence is only shown at level 3.}\fi
		\fi
	\end{enumerate}
\end{document}
